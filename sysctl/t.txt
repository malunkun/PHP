    function setDHCPDNS($dns){
        $file = fopen(self::conffile,"r");
        while(!feof($file)){
            $filestr = fgets($file);
            if(stristr($filestr,"server=")){
                $olddns =  $filestr;
            }
        }
        fclose($file);
        $allFile = file_get_contents(self::conffile);
        $allFile = str_replace($olddns,"server=".$dns."\n",$allFile);
        file_put_contents(self::conffile,$allFile);
        return true;
    }
